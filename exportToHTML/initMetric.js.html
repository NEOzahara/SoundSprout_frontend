import { generateUniqueID } from './generateUniqueID.js';
export const initMetric = (name, value) => {
    return {
        name,
        value: typeof value === 'undefined' ? -1 : value,
        delta: 0,
        entries: [],
        id: generateUniqueID()
    };
};
